#include <iostream>
#include <cstdlib>  
#include <ctime>    
using namespace std;

void generateKey(int index) {
    cout << "Key [" << index << "]: ";

    for (int group = 0; group < 4; group++) {
        for (int i = 0; i < 4; i++) {
            char letter = 'A' + rand() % 26; 
            cout << letter;
        }

        if (group < 3) {
            cout << "-"; 
        }
    }

    cout << endl;
}

int main() {
    int n;
    cout << "Enter number of keys: ";
    cin >> n;

    srand(time(0)); 
    
    for (int i = 1; i <= n; i++) {
        generateKey(i);
    }

    return 0;
}
