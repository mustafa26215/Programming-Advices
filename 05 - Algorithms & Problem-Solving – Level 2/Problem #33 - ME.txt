#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    string arr[N];
    
    srand(time(0));
    
    for (int i = 0; i < N; i++) {
        string key = "";
        for (int j = 0; j < 16; j++) {
            char c = 'A' + rand() % 26;
            key += c;
            if (j == 3 || j == 7 || j == 11) {
                key += '-';
            }
        }
        arr[i] = key;
    }
    
    cout << "Array elements:" << endl;
    for (int i = 0; i < N; i++) {
        cout << "Array[" << i << "]: " << arr[i] << endl;
    }
    
    return 0;
}
