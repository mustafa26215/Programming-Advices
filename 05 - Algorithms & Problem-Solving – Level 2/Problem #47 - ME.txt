#include <iostream>
#include <cmath>
using namespace std;

int main() {
    double n;
    cin >> n;
    
    int myRound = (n >= 0) ? (int)(n + 0.5) : (int)(n - 0.5);
    
    cout << "My Round Result : " << myRound << endl;
    cout << "C++ Round Result: " << (int)round(n) << endl;
    
    return 0;
}
