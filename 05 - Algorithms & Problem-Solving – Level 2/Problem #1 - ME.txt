#include <iostream>
using namespace std;

void PrintHeaderOfTable() {

    cout << "\n\t\tMultiplication Table From 1 to 10\n\n";

    cout << "    ";   

    for (int i = 1; i <= 10; i++) {
        cout << i << "    "; 
    }

    cout << "\n----------------------------------------------------------------\n";
}

void TableOftMultiplication() {

    for (int i = 1; i <= 10; i++) {

        cout << i << " |   ";

        for (int j = 1; j <= 10; j++) {
            cout << i * j << "    ";
        }

        cout << endl;
    }
}

int main()
{
    PrintHeaderOfTable();
    TableOftMultiplication();
}
