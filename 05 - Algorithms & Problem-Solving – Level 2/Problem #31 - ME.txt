#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    int arr[N];
    
    for (int i = 0; i < N; i++) {
        arr[i] = i + 1;
    }
    
    cout << "Array elements before shuffle:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    srand(time(0));
    
    for (int i = N - 1; i > 0; i--) {
        int j = rand() % (i + 1);
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
    
    cout << "Array elements after shuffle:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    return 0;
}
