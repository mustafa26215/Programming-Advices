#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

void fillArray(int arr[], int n) {
    for(int i = 0; i < n; i++)
        arr[i] = rand() % 100 + 1;
}

void printArray(int arr[], int n, string name) {
    cout << name << ": ";
    for(int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl;
}

void sumArrays(int arr1[], int arr2[], int arrSum[], int n) {
    for(int i = 0; i < n; i++)
        arrSum[i] = arr1[i] + arr2[i];
}

int main() {
    int n, arr1[100], arr2[100], arrSum[100];
    cin >> n;
    if(n > 100 || n <= 0) return 1;

    srand(time(0));

    fillArray(arr1, n);
    fillArray(arr2, n);
    sumArrays(arr1, arr2, arrSum, n);

    printArray(arr1, n, "Array 1 elements");
    printArray(arr2, n, "Array 2 elements");
    printArray(arrSum, n, "Sum of array1 and array2 elements");

    return 0;
}
