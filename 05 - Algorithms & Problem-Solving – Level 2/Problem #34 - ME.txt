#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    int arr[100];
    
    srand(time(0));
    
    for (int i = 0; i < N; i++) {
        arr[i] = rand() % 100 + 1;
    }
    
    cout << "Array 1 elements:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    int searchNum;
    cout << "Please enter a number to search for?" << endl;
    cin >> searchNum;
    
    cout << "Number you are looking for is: " << searchNum << endl;
    
    int index = -1;
    for (int i = 0; i < N; i++) {
        if (arr[i] == searchNum) {
            index = i;
            break;
        }
    }
    
    if (index != -1) {
        cout << "The number found at position: " << index << endl;
        cout << "The number found its order : " << index + 1 << endl;
    } else {
        cout << "The number is not found :-(" << endl;
    }
    
    return 0;
}
