#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));

    int numQuestions = 5;
    int rightAnswers = 0, wrongAnswers = 0;

    for (int i = 1; i <= numQuestions; i++) {
        int num1 = rand() % 10 + 1;
        int num2 = rand() % 10 + 1;
        int op   = rand() % 4 + 1;

        string symbol;
        int correctAnswer;

        if (op == 1) { symbol = "+"; correctAnswer = num1 + num2; }
        else if (op == 2) { symbol = "-"; correctAnswer = num1 - num2; }
        else if (op == 3) { symbol = "x"; correctAnswer = num1 * num2; }
        else             { symbol = "/"; correctAnswer = num1 / num2; }

        cout << "\nQuestion [" << i << "/" << numQuestions << "]\n";
        cout << num1 << " " << symbol << " " << num2 << " = ";

        int playerAnswer;
        cin >> playerAnswer;

        if (playerAnswer == correctAnswer) {
            cout << "Correct!\n";
            rightAnswers++;
        } else {
            cout << "Wrong! Correct Answer: " << correctAnswer << "\n";
            wrongAnswers++;
        }
    }

    cout << "\n===== Quiz Results =====\n";
    cout << "Right Answers: " << rightAnswers << "\n";
    cout << "Wrong Answers: " << wrongAnswers << "\n";

    if (rightAnswers >= wrongAnswers)
        cout << "You Passed the Quiz!\n";
    else
        cout << "You Failed the Quiz!\n";

    return 0;
}
