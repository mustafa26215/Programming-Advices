#include <iostream>
using namespace std;

void AddArrayElement(int arr[], int &length, int num) {
    arr[length] = num;
    length++;
}

bool isNumberInArray(int arr[], int length, int num) {
    for (int i = 0; i < length; i++) {
        if (arr[i] == num) return true;
    }
    return false;
}

int main() {
    int arr1[] = {10, 10, 10, 50, 50, 70, 70, 70, 70, 90};
    int size1 = 10;
    
    int arr2[100];
    int length2 = 0;
    
    cout << "Array 1 elements:" << endl;
    for (int i = 0; i < size1; i++) {
        cout << arr1[i] << " ";
    }
    cout << endl;
    
    for (int i = 0; i < size1; i++) {
        if (!isNumberInArray(arr2, length2, arr1[i])) {
            AddArrayElement(arr2, length2, arr1[i]);
        }
    }
    
    cout << "Array 2 distinct elements:" << endl;
    for (int i = 0; i < length2; i++) {
        cout << arr2[i] << " ";
    }
    cout << endl;
    
    return 0;
}
