#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

bool isPrime(int n) {
    if(n < 2) return false;
    for(int i = 2; i*i <= n; i++)
        if(n % i == 0) return false;
    return true;
}

void fillArray(int arr[], int n) {
    for(int i = 0; i < n; i++)
        arr[i] = rand() % 100 + 1;
}

void printArray(int arr[], int n, string name) {
    cout << name << ": ";
    for(int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << endl;
}

int main() {
    int n, arr1[100], arr2[100], primeCount = 0;
    cin >> n;
    if(n > 100 || n <= 0) return 1;

    srand(time(0));

    fillArray(arr1, n);

    for(int i = 0; i < n; i++)
        if(isPrime(arr1[i]))
            arr2[primeCount++] = arr1[i];

    printArray(arr1, n, "Array 1 elements");
    printArray(arr2, primeCount, "Prime Numbers in Array2");

    return 0;
}
