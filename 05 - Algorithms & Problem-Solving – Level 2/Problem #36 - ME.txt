#include <iostream>
using namespace std;

int main() {
    int arr[100];
    int length = 0;
    int num;
    int choice;
    
    do {
        cout << "Please enter a number? ";
        cin >> num;
        arr[length] = num;
        length++;
        
        cout << "Do you want to add more numbers? [0]:No,[1]:yes? ";
        cin >> choice;
    } while (choice == 1 && length < 100);
    
    cout << "Array Length: " << length << endl;
    cout << "Array elements: ";
    for (int i = 0; i < length; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    return 0;
}
