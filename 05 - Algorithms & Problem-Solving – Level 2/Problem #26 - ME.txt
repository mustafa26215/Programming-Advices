#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main() {
    int n;
    int arr[100];
    int sum = 0;

    cin >> n;

    if(n < 1 || n > 100) return 1;

    srand(time(0));

    for(int i = 0; i < n; i++) {
        arr[i] = rand() % 100 + 1;
        cout << arr[i] << " ";
        sum += arr[i];
    }
    cout << endl;
    cout << "Sum of all number is: " << sum << endl;

    return 0;
}
