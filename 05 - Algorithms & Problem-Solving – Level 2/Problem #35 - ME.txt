#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    int arr[100];
    
    srand(time(0));
    
    for (int i = 0; i < N; i++) {
        arr[i] = rand() % 100 + 1;
    }
    
    cout << "Array 1 elements:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    int searchNum;
    cout << "Please enter a number to search for?" << endl;
    cin >> searchNum;
    
    cout << "Number you are looking for is: " << searchNum << endl;
    
    bool found = false;
    for (int i = 0; i < N; i++) {
        if (arr[i] == searchNum) {
            found = true;
            break;
        }
    }
    
    if (found) {
        cout << "Yes, The number is found :-)" << endl;
    } else {
        cout << "No, The number is not found :-(" << endl;
    }
    
    return 0;
}
