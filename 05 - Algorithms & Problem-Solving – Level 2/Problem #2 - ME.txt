#include <iostream>
using namespace std;

int main()
{
    int a;
    cin >> a;
                     // a = 10
    for (int i = 2; i <= a; i++)
    {
        int count = 0; //1

        for (int j = 1; j <= i; j++)
        {
            if (i % j == 0) //2 % 1|| 2 % 2 but 2 % 3 
                count++;
        }

        if (count == 2)
            cout << i << endl;
    }
}
