#include <iostream>
using namespace std;

int main()
{
    int a;
    cin >> a;  

    for (int num = 1; num <= a; num++)   
    {
        int Sum = 0;   

        for (int i = 1; i < num; i++)  
        {
            if (num % i == 0)
                Sum += i;
        }

        if (Sum == num)   
            cout << num << endl;
    }
}
