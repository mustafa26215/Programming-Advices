#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

void AddArrayElement(int arr[], int &length, int num) {
    arr[length] = num;
    length++;
}

bool isPrime(int num) {
    if (num <= 1) return false;
    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0) return false;
    }
    return true;
}

int main() {
    int N;
    cin >> N;
    
    int arr1[100];
    int arr2[100];
    int length2 = 0;
    
    srand(time(0));
    
    for (int i = 0; i < N; i++) {
        arr1[i] = rand() % 100 + 1;
    }
    
    cout << "Array 1 elements:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr1[i] << " ";
    }
    cout << endl;
    
    for (int i = 0; i < N; i++) {
        if (isPrime(arr1[i])) {
            AddArrayElement(arr2, length2, arr1[i]);
        }
    }
    
    cout << "Array 2 Prime numbers:" << endl;
    for (int i = 0; i < length2; i++) {
        cout << arr2[i] << " ";
    }
    cout << endl;
    
    return 0;
}
