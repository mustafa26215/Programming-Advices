#include <iostream>
using namespace std;

int main()
{
   
    int a;
    cin >> a;
    int Sum = 0;
    for (int i = 1; i < a; i++) {
       
        if ( a% i == 0) {
            Sum=Sum+i;
        }
      
    }
    if (Sum == a) {
        cout << "Perfect Number";
    }
    else {
        cout << "Not";
    }
}
