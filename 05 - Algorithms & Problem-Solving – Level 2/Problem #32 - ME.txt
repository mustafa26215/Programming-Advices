#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    int N;
    cin >> N;
    
    int arr1[100];
    int arr2[100];
    
    srand(time(0));
    
    for (int i = 0; i < N; i++) {
        arr1[i] = rand() % 100 + 1;
    }
    
    cout << "Array 1 elements:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr1[i] << " ";
    }
    cout << endl;
    
    for (int i = 0; i < N; i++) {
        arr2[i] = arr1[N - 1 - i];
    }
    
    cout << "Array 2 elements after copying array 1 in reversed order:" << endl;
    for (int i = 0; i < N; i++) {
        cout << arr2[i] << " ";
    }
    cout << endl;
    
    return 0;
}
