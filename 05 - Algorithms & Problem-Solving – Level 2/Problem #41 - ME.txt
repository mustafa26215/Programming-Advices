#include <iostream>
using namespace std;

int main() {
    int arr[100];
    int length;
    
    cin >> length;
    
    for (int i = 0; i < length; i++) {
        cin >> arr[i];
    }
    
    cout << "Array Elements:" << endl;
    for (int i = 0; i < length; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    bool isPalindrome = true;
    for (int i = 0; i < length / 2; i++) {
        if (arr[i] != arr[length - 1 - i]) {
            isPalindrome = false;
            break;
        }
    }
    
    if (isPalindrome) {
        cout << "Yes array is Palindrome" << endl;
    } else {
        cout << "No array is not Palindrome" << endl;
    }
    
    return 0;
}
