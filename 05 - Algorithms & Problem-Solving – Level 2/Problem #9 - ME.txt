#include <iostream>
using namespace std;

int main()
{
    int a;
    cin >> a;

    int freq[10] = {0};

    while (a > 0) {
        int remainder = a % 10;

        freq[remainder]++;   

        a /= 10;
    }


    for (int i = 0; i < 10; i++) {
        if (freq[i] > 0) {
            cout << "Digit " << i << " Frequency is " 
                 << freq[i] << " Time(s)." << endl;
        }
    }
}
