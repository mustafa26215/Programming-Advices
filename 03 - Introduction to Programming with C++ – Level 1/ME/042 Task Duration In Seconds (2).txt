#include <iostream>
#include <cmath>
using namespace std;

int main() {
    double NumberOfDays, NumberOfHours, NumberOfMinutes, NumberOfSeconds;
    double TotalSeconds;

    cin >> NumberOfDays;
    cin >> NumberOfHours;
    cin >> NumberOfMinutes;
    cin >> NumberOfSeconds;

    TotalSeconds = (NumberOfDays * 24 * 60 * 60) +
                   (NumberOfHours * 60 * 60) +
                   (NumberOfMinutes * 60) +
                   NumberOfSeconds;

    cout << round(TotalSeconds) << " Seconds";

    return 0;
}
