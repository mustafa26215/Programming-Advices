#include <iostream>
using namespace std;

int main()
{
    int PIN;
    int Balance = 7500;

    do
    {
        cout << "Enter ATM PIN: ";
        cin >> PIN;

        if (PIN != 1234)
            cout << "Wrong PIN\n";

    } while (PIN != 1234);

    cout << "Your Balance is: " << Balance;

    return 0;
}
