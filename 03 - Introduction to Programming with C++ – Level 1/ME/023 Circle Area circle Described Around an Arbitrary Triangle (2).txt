#include <iostream>
#include <cmath>
using namespace std;

// Function to calculate circle area described around a triangle
double CalculateCircumcircleArea(double a, double b, double c)
{
    const double PI = 3.14;

    double p = (a + b + c) / 2;

    double R = (a * b * c) / (4 * sqrt(p * (p - a) * (p - b) * (p - c)));

    return PI * R * R;
}

int main()
{
    double a, b, c, area;

    cout << "Enter a: ";
    cin >> a;

    cout << "Enter b: ";
    cin >> b;

    cout << "Enter c: ";
    cin >> c;

    area = CalculateCircumcircleArea(a, b, c);

    cout << "Circle Area = " << area;

    return 0;
}
