#include <iostream>
using namespace std;

int main()
{
    int day;

    // repeat until user enters valid day
    do
    {
        cout << "Enter Day (1-7): ";
        cin >> day;

        if (day == 1)
            cout << "It's Sunday" << endl;

        else if (day == 2)
            cout << "It's Monday" << endl;

        else if (day == 3)
            cout << "It's Tuesday" << endl;

        else if (day == 4)
            cout << "It's Wednesday" << endl;

        else if (day == 5)
            cout << "It's Thursday" << endl;

        else if (day == 6)
            cout << "It's Friday" << endl;

        else if (day == 7)
            cout << "It's Saturday" << endl;

        else
            cout << "Wrong Day, try again!" << endl;

    } while (day < 1 || day > 7);

    return 0;
}
