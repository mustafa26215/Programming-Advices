#include <iostream>
using namespace std;

int main()
{
    int Num, M;
    int P = 1;

    cout << "Enter Number: ";
    cin >> Num;

    cout << "Enter Power (M): ";
    cin >> M;

    for (int i = 0; i < M; i++)
    {
        P = P * Num;
    }

    cout << "Result = " << P << endl;

    return 0;
}
