#include <iostream>
using namespace std;

struct Person {
    string firstName;
    string lastName;
    int age;
    string phone;
};

void input(Person p[], int n){
    for(int i=0;i<n;i++){
        cout<<"Person "<<i+1<<" Info\n";
        cout<<"First Name: ";
        cin>>p[i].firstName;
        cout<<"Last Name: ";
        cin>>p[i].lastName;
        cout<<"Age: ";
        cin>>p[i].age;
        cout<<"Phone: ";
        cin>>p[i].phone;
    }
}

void output(Person p[], int n){
    for(int i=0;i<n;i++){
        cout<<"\nPerson "<<i+1<<" Info\n";
        cout<<"First Name: "<<p[i].firstName<<endl;
        cout<<"Last Name: "<<p[i].lastName<<endl;
        cout<<"Age: "<<p[i].age<<endl;
        cout<<"Phone: "<<p[i].phone<<endl;
    }
}

int main(){
    int n;
    cin>>n;

    Person p[n];

    input(p,n);
    output(p,n);

    return 0;
}
