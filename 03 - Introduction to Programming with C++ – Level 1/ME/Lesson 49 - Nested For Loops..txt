#include <iostream>
using namespace std;

int main() {

    for(char i='A'; i<='Z'; i++){
        for(char j='A'; j<='Z'; j++){
            cout << i << j << endl;
        }
    }
    cout << endl;

    for(int i=10;i>=1;i--){
        for(int j=1;j<=i;j++) cout<<"*";
        cout<<endl;
    }
    cout << endl;

    for(int i=10;i>=1;i--){
        for(int j=1;j<=i;j++) cout<<j;
        cout<<endl;
    }
    cout << endl;

    for(int i=1;i<=10;i++){
        for(int j=1;j<=i;j++) cout<<j;
        cout<<endl;
    }
    cout << endl;

    for(char i='A'; i<='F'; i++){
        for(char j='A'; j<=i; j++) cout<<j;
        cout<<endl;
    }
    cout << endl;

    for(int i=1;i<=10;i++){
        for(int j=i;j<=10;j++) cout<<j;
        cout<<endl;
    }

    return 0;
}
