#include <iostream>
using namespace std;

int main()
{
    int Day;

    do
    {
        cout << "Enter day number (1-7): ";
        cin >> Day;

        if (Day < 1 || Day > 7)
            cout << "Wrong Day, try again.\n";

    } while (Day < 1 || Day > 7);

    switch (Day)
    {
        case 1:
            cout << "It's Sunday";
            break;

        case 2:
            cout << "It's Monday";
            break;

        case 3:
            cout << "It's Tuesday";
            break;

        case 4:
            cout << "It's Wednesday";
            break;

        case 5:
            cout << "It's Thursday";
            break;

        case 6:
            cout << "It's Friday";
            break;

        case 7:
            cout << "It's Saturday";
            break;
    }

    return 0;
}
