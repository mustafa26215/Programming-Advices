#include <iostream>
using namespace std;

int main()
{
    int days, hours, minutes, seconds;
    int totalSeconds;

    cout << "Enter number of days: ";
    cin >> days;

    cout << "Enter number of hours: ";
    cin >> hours;

    cout << "Enter number of minutes: ";
    cin >> minutes;

    cout << "Enter number of seconds: ";
    cin >> seconds;

    totalSeconds =
        (days * 24 * 60 * 60) +
        (hours * 60 * 60) +
        (minutes * 60) +
        seconds;

    cout << totalSeconds << " Seconds";

    return 0;
}
