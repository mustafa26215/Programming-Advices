#include <iostream>
using namespace std;


struct Person {
    string name;
    int age;
    string city;
    string country;
    double monthlySalary;
    double yearlySalary;
    char gender;
    bool married;
};


void printPersonInfo(Person p) {

    cout<<"************************\n";
    cout<<"Name: " << p.name << endl;
    cout<<"Age: " << p.age << " Years\n";
    cout<<"City: " << p.city << endl;
    cout<<"Country: " << p.country << endl;
    cout<<"Monthly Salary: " << p.monthlySalary << endl;
    cout<<"Yearly Salary: " << p.yearlySalary << endl;
    cout<<"Gender: " << p.gender << endl;
    cout<<"Married: " << p.married << endl;
    cout<<"************************\n";
}

int main() {

    Person p1;

    p1.name = "Mohammed Abu-Hadhoud";
    p1.age = 44;
    p1.city = "Amman";
    p1.country = "Jordan";
    p1.monthlySalary = 5000;
    p1.yearlySalary = 60000;
    p1.gender = 'M';
    p1.married = true;

    printPersonInfo(p1); 
    return 0;
}
